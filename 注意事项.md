```jsx
useEffect(
  () => {
    // 1. 副作用函数：执行你的逻辑
    // 2. 可选：返回一个清理函数（清除副作用）
    return () => {
      /* 清理逻辑 */
    };
  },
  [
    /* 3. 依赖项数组：控制副作用执行时机 */
  ],
);

// 1. 无依赖项 → 每次渲染都执行
// 2. 空依赖项 [] → 仅首次渲染执行（最常用）
// 3. 带具体依赖项 [state, props] → 依赖变化时执行

// 清理计时器:
useEffect(() => {
  // 1. 创建定时器
  const timer = setInterval(() => {
    console.log("定时器执行");
  }, 1000);

  // 2. 返回清理函数：组件卸载/重新执行前清除定时器
  return () => {
    clearInterval(timer);
    console.log("定时器已清除");
  };
}, []);

// 4. 不能传入异步函数
```
